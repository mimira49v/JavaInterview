
public static int unique(int[] x){
Set<Integer>uniqueNumber=new HashSet<>();
Set<Integer>repeatedNumbers=new HashSet<>();

for(int e:x){
    if(reapeatedNumbers.contains(e)){
        continue;
    }
    if(uniqueNumber.contains(e)){
        uniqueNumber.remove(e);
        repeatedNumbers.add(e);
    }else{
        uniqueNumber.add(e);
    }
    for(int e:x)}{
        if(uniqueNumber.contains(e)){
        return e;
        }
    }
    return 0;
}
}
